// Test the conflict detection API directly
async function testConflictAPI(start, end, description) {
    console.log(`\n🧪 Testing: ${description}`);
    console.log(`   Time: ${start} to ${end}`);
    
    try {
        const response = await fetch('/events/api/check-conflicts', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-Requested-With': 'XMLHttpRequest'
            },
            credentials: 'same-origin',
            body: JSON.stringify({
                start: start,
                end: end
            })
        });
        
        console.log(`   Response status: ${response.status}`);
        
        if (response.ok) {
            const result = await response.json();
            console.log('   Result:', result);
            
            if (result.success) {
                if (result.has_conflicts) {
                    console.log(`   ⚠️  CONFLICTS DETECTED (${result.conflict_count} conflicts)`);
                    if (result.conflicts) {
                        result.conflicts.forEach((conflict, i) => {
                            console.log(`     ${i+1}. ${conflict.title} (${conflict.start} to ${conflict.end})`);
                        });
                    }
                } else {
                    console.log('   ✅ NO CONFLICTS - Time slot available');
                }
            } else {
                console.log(`   ❌ API Error: ${result.message}`);
            }
        } else {
            console.log(`   ❌ HTTP Error: ${response.status}`);
            const text = await response.text();
            console.log('   Response:', text.substring(0, 200) + '...');
        }
    } catch (error) {
        console.log(`   ❌ Network Error: ${error.message}`);
    }
}

// Test modal conflict detection with Feb 5th data
async function testModalWithFeb5Data() {
    console.log('\n🗓️  TESTING WITH FEBRUARY 5TH DATA');
    console.log('===================================');
    
    const elements = testModalFunctions();
    
    if (!elements.startInput || !elements.endInput) {
        console.log('❌ Modal not open. Opening modal...');
        if (typeof window.openEventModal === 'function') {
            window.openEventModal();
            await new Promise(resolve => setTimeout(resolve, 1000));
            return testModalWithFeb5Data(); // Retry after opening modal
        } else {
            console.log('❌ Cannot open modal. Please open it manually.');
            return;
        }
    }
    
    // Test scenarios for Feb 5th
    const testCases = [
        {
            name: 'Overlap with Test Event',
            start: '2026-02-05T10:15',
            end: '2026-02-05T10:45',
            expected: 'CONFLICT'
        },
        {
            name: 'Clear morning slot',
            start: '2026-02-05T09:00',
            end: '2026-02-05T09:30',
            expected: 'CLEAR'
        },
        {
            name: 'Overlap with multiple events',
            start: '2026-02-05T14:30',
            end: '2026-02-05T15:30',
            expected: 'CONFLICT'
        }
    ];
    
    for (const testCase of testCases) {
        console.log(`\n📝 Testing: ${testCase.name}`);
        
        // Set the input values
        elements.startInput.value = testCase.start;
        elements.endInput.value = testCase.end;
        
        console.log(`   Set times: ${testCase.start} to ${testCase.end}`);
        
        // Trigger conflict detection
        if (typeof window.checkAndDisplayConflicts === 'function') {
            console.log('   Triggering checkAndDisplayConflicts...');
            await window.checkAndDisplayConflicts();
            
            // Wait for the API call to complete
            await new Promise(resolve => setTimeout(resolve, 2000));
            
            // Check the conflict status display
            if (elements.conflictStatus) {
                const statusHTML = elements.conflictStatus.innerHTML;
                console.log('   Conflict status updated:', !statusHTML.includes('Checking'));
                
                if (statusHTML.includes('Conflict Detected')) {
                    console.log('   ⚠️  Modal shows: CONFLICTS DETECTED');
                } else if (statusHTML.includes('No Conflicts')) {
                    console.log('   ✅ Modal shows: NO CONFLICTS');
                } else if (statusHTML.includes('Checking')) {
                    console.log('   ⏳ Modal shows: Still checking...');
                } else {
                    console.log('   ❓ Modal shows: Unknown state');
                    console.log('   Status HTML:', statusHTML.substring(0, 100) + '...');
                }
            }
        } else {
            console.log('   ❌ checkAndDisplayConflicts function not available');
        }
        
        console.log('   ----------------------------------------');
    }
}
